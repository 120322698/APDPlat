30      @Override
31      public void doFilter(ServletRequest request, ServletResponse response, FilterChain chain) throws IOException,
32              ServletException {
33          HttpServletRequest req=(HttpServletRequest)request;
34          
35          long start=0;
36          if (enabled && filter(req)) {            
37  		start=System.currentTimeMillis();
38          }
39          chain.doFilter(request, response);
40          if (enabled && filter(req)) {
41  		long end=System.currentTimeMillis();
42                  User user=OnlineUserService.getUser(req.getSession().getId());
43                  ProcessTime logger=new ProcessTime();
44                  logger.setOwnerUser(user);
45                  logger.setUserIP(req.getRemoteAddr());
46                  try {
47                      logger.setServerIP(InetAddress.getLocalHost().getHostAddress());
48                  } catch (UnknownHostException ex) {
49                      log.error("保存日志出错(Error in saving log)",ex);
50                  }
51                  logger.setAppName(SystemListener.getContextPath());
52                  String resource=req.getRequestURI().replace(logger.getAppName(), "");
53                  logger.setResource(resource);
54                  logger.setStartTime(new Date(start));
55                  logger.setEndTime(new Date(end));
56                  logger.setProcessTime(end-start);
57                  LogQueue.addLog(logger);
58          }
59      }

77      private boolean filter(HttpServletRequest req) {
78          String path=req.getRequestURI();
79          if(path.contains("/log/")){
80              log.info("路径包含/log/,不执行性能分析(/log/ in path, not execute performance analysis) "+path);
81              return false;
82          }
83          if(path.contains("/monitor/")){
84              log.info("路径包含/monitor/,不执行性能分析(/log/ in path, not execute performance analysis) "+path);
