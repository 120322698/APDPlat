47      /**

49       */
50      @PostConstruct
51      public  void initSecurityConfigInfo(){
52          String security=PropertyHolder.getProperty("security");
53          if(security==null || !"true".equals(security.trim())){
54              log.info("当前系统禁用安全机制");
55              return ;
56          }
57          log.info("开始初始化权限子系统...");
58          LinkedHashMap<RequestKey, Collection<ConfigAttribute>> requestMap =new LinkedHashMap<>();
59          
60          SecurityConfig manager=new SecurityConfig("ROLE_MANAGER");
61          SecurityConfig superManager=new SecurityConfig("ROLE_SUPERMANAGER");
62          Collection<ConfigAttribute> value=new ArrayList<>();
63          value.add(manager);
64          value.add(superManager);
65          Collection<String> urls=new LinkedHashSet<>(); 
66          String[] urlFiles=PropertyHolder.getProperty("manager.default.url").split(",");
67          for(String urlFile : urlFiles){
68              Collection<String> url=FileUtils.getClassPathTextFileContent(urlFile);
69              urls.addAll(url);
70          }
71          for(String url : urls){
72              if(url.contains("=")){
73                  String[] attr=url.split("=");
74                  url=attr[0];
75                  String[] roles=attr[1].split(",");
76                  Collection<ConfigAttribute> v=new ArrayList<>();
77                  for(String role : roles){
78                      v.add(new SecurityConfig(role));
79                  }
80                  //POST
81                  RequestKey key=new RequestKey(url,"POST");
82                  requestMap.put(key, v);
83                  //GET
84                  key=new RequestKey(url,"GET");
85                  requestMap.put(key, v);
86              }else{
87                  //POST
88                  RequestKey key=new RequestKey(url,"POST");
89                  requestMap.put(key, value);
90                  //GET
91                  key=new RequestKey(url,"GET");
92                  requestMap.put(key, value);
93              }
94          }
95  
96          for(Command command : serviceFacade.query(Command.class).getModels()){
97              List<String> paths=ModuleService.getCommandPath(command);
98              Map<String,String> map=ModuleService.getCommandPathToRole(command);
99              for(String path : paths){
100                 RequestKey key=new RequestKey(path.toString().toLowerCase()+".action*","POST");
101                 value=new ArrayList<>();
102                 value.add(new SecurityConfig("ROLE_MANAGER"+map.get(path)));
103                 value.add(superManager);
104                 requestMap.put(key, value);
105                 //GET
106                 key=new RequestKey(path.toString().toLowerCase()+".action*","GET");
107                 requestMap.put(key, value);
108             }
109         }
110         RequestKey key=new RequestKey("/**","POST");
111         value=new ArrayList<>();
112         value.add(superManager);
113         requestMap.put(key, value);
114         //GET
115         key=new RequestKey("/**","GET");
116         requestMap.put(key, value);        
117 
118         DefaultFilterInvocationSecurityMetadataSource source=new DefaultFilterInvocationSecurityMetadataSource(new AntUrlPathMatcher(),requestMap);
119         
120         filterSecurityInterceptor.setSecurityMetadataSource(source);
121 
122         log.debug("system privilege info:\n");
123         for(Map.Entry<RequestKey, Collection<ConfigAttribute>> entry : requestMap.entrySet()){
124             log.debug(entry.getKey().toString());
125             for(ConfigAttribute att : entry.getValue()){
126                 log.debug("\t"+att.toString());
127             }
128         }
129         log.info("完成初始化权限子系统...");
130     }
