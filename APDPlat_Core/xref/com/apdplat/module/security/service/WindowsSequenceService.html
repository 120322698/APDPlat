27       * @param drive 硬盘驱动器分区 如C,D

30      private String getHDSerial(String drive) {
31          String result = "";
32          try {
33              File file = File.createTempFile("tmp", ".vbs");
34              file.deleteOnExit();
35              try (FileWriter fw = new java.io.FileWriter(file)) {
36                  String vbs = "Set objFSO = CreateObject(\"Scripting.FileSystemObject\")\n"
37                          + "Set colDrives = objFSO.Drives\n" + "Set objDrive = colDrives.item(\"" + drive + "\")\n"
38                          + "Wscript.Echo objDrive.SerialNumber";
39                  fw.write(vbs);
40              }
41              Process p = Runtime.getRuntime().exec("cscript //NoLogo " + file.getPath());
42              try (BufferedReader input = new BufferedReader(new InputStreamReader(p.getInputStream()))) {
43                  String line;
44                  while ((line = input.readLine()) != null) {
45                      result += line;
46                  }
47              }
48              file.delete();
49          } catch (Throwable e) {
50              log.error("生成HDSerial失败", e);
51          }
52          if (result.trim().length() < 1 || result == null) {
53              log.info("无磁盘ID被读取");
54              result = "";
55          }
56  
57          return result.trim();
58      }
