28              if(node==null){
29                  return super.query();

32                  String json=moduleService.toRootJsonForEdit();
33                  Struts2Utils.renderJson(json);
34                  return null;
35              }
36              
37              if(node.contains("-")){
38                  try{
39                      String[] temp=node.split("-");
40                      int id=Integer.parseInt(temp[1]);
41                      Module module=moduleService.getModule(id);
42                      String json=moduleService.toJsonForEdit(module);
43                      Struts2Utils.renderJson(json);
44                  }catch(Exception e){
45                      log.error("获取根模块出错",e);
46                  }
47              }
