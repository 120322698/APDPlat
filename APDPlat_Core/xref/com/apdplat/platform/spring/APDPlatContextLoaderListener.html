30  /**
31   * Bootstrap listener to start up and shut down Spring's root {@link WebApplicationContext}.
32   * Simply delegates to {@link ContextLoader} as well as to {@link ContextCleanupListener}.
33   *
34   * <p>This listener should be registered after
35   * {@link org.springframework.web.util.Log4jConfigListener}
36   * in <code>web.xml</code>, if the latter is used.
37   *
38   * @author Juergen Hoeller
39   * @since 17.02.2003
40   * @see org.springframework.web.util.Log4jConfigListener
41   */
42  public class APDPlatContextLoaderListener extends ContextLoader implements ServletContextListener {
43  
44  	private ContextLoader contextLoader;
45  
46  
47  	/**
48  	 * Initialize the root web application context.
49  	 */
50  	public void contextInitialized(ServletContextEvent event) {
51              SystemListener.contextInitialized(event);
52  		this.contextLoader = createContextLoader();
53  		if (this.contextLoader == null) {
54  			this.contextLoader = this;
55  		}
56  		this.contextLoader.initWebApplicationContext(event.getServletContext());
57  	}
58  
59  	/**
60  	 * Create the ContextLoader to use. Can be overridden in subclasses.
61  	 * @return the new ContextLoader
62  	 * @deprecated in favor of simply subclassing APDPlatContextLoaderListener itself
63  	 * (which extends ContextLoader, as of Spring 3.0)
64  	 */
65  	@Deprecated
66  	protected ContextLoader createContextLoader() {
67  		return null;
68  	}
69  
70  	/**
71  	 * Return the ContextLoader used by this listener.
72  	 * @return the current ContextLoader
73  	 * @deprecated in favor of simply subclassing APDPlatContextLoaderListener itself
74  	 * (which extends ContextLoader, as of Spring 3.0)
75  	 */
76  	@Deprecated
77  	public ContextLoader getContextLoader() {
78  		return this.contextLoader;
79  	}
80  
81  
82  	/**
83  	 * Close the root web application context.
84  	 */
85  	public void contextDestroyed(ServletContextEvent event) {
86              SystemListener.contextDestroyed(event);
87  		if (this.contextLoader != null) {
88  			this.contextLoader.closeWebApplicationContext(event.getServletContext());
89  		}
90  		ContextCleanupListener.cleanupAttributes(event.getServletContext());
91  	}
92  
93  }
94  class ContextCleanupListener implements ServletContextListener {
95  
96  	private static final Log logger = LogFactory.getLog(ContextCleanupListener.class);
97  
98  
99  	public void contextInitialized(ServletContextEvent event) {
100 	}
101 
102 	public void contextDestroyed(ServletContextEvent event) {
103 		cleanupAttributes(event.getServletContext());
104 	}
105 
106 
107 	/**
108 	 * Find all ServletContext attributes which implement {@link DisposableBean}
109 	 * and destroy them, removing all affected ServletContext attributes eventually.
110 	 * @param sc the ServletContext to check
111 	 */
112 	static void cleanupAttributes(ServletContext sc) {
113 		Enumeration attrNames = sc.getAttributeNames();
114 		while (attrNames.hasMoreElements()) {
115 			String attrName = (String) attrNames.nextElement();
116 			if (attrName.startsWith("org.springframework.")) {
117 				Object attrValue = sc.getAttribute(attrName);
118 				if (attrValue instanceof DisposableBean) {
119 					try {
120 						((DisposableBean) attrValue).destroy();
121 					}
122 					catch (Throwable ex) {
123 						logger.error("Couldn't invoke destroy method of attribute with name '" + attrName + "'", ex);
124 					}
125 				}
126 			}
127 		}
128 	}
