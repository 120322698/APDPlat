30      /**
31       * 对一个文件求他的md5值
32       * @param f 要求md5值的文件
33       * @return md5串
34       */
35      public static String md5(File f) {
36          FileInputStream fis = null;
37          try {
38              fis = new FileInputStream(f);
39              byte[] buffer = new byte[8192];
40              int length;
41              while ((length = fis.read(buffer)) != -1) {
42                  md.update(buffer, 0, length);
43              }
44  
45              return new String(Hex.encodeHex(md.digest()));
46          } catch (FileNotFoundException e) {
47              log.error("md5 file " + f.getAbsolutePath() + " failed:" + e.getMessage());
48              return null;
49          } catch (IOException e) {
50              log.error("md5 file " + f.getAbsolutePath() + " failed:" + e.getMessage());
51              return null;
52          } finally {
53              try {
54                  if (fis != null) {
55                      fis.close();
56                  }
57              } catch (IOException ex) {
58               log.error("文件操作失败",ex);
59              }
60          }
61      }
62  
63      /**
64       * 求一个字符串的md5值
65       * @param target 字符串
66       * @return md5 value
67       */
68      public static String md5(String target) {
69          return DigestUtils.md5Hex(target);
