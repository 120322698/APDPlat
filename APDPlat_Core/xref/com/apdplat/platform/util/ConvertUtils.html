15      public static String getTimeDes(long ms) {
16          int ss = 1000;
17          int mi = ss * 60;
18          int hh = mi * 60;
19          int dd = hh * 24;
20  
21          long day = ms / dd;
22          long hour = (ms - day * dd) / hh;
23          long minute = (ms - day * dd - hour * hh) / mi;
24          long second = (ms - day * dd - hour * hh - minute * mi) / ss;
25          long milliSecond = ms - day * dd - hour * hh - minute * mi - second * ss;
26          
27          StringBuilder str=new StringBuilder();
28          if(day>0){
29              str.append(day).append("天,");
30          }
31          if(hour>0){
32              str.append(hour).append("小时,");
33          }
34          if(minute>0){
35              str.append(minute).append("分钟,");
36          }
37          if(second>0){
38              str.append(second).append("秒,");
39          }
40          if(milliSecond>0){
41              str.append(milliSecond).append("毫秒,");
42          }
43          if(str.length()>0){
44              str=str.deleteCharAt(str.length()-1);
45          }
46  
47          return str.toString();
48      }

50      public static String byte2HexString(byte[] b) {
51          char[] hex = {'0', '1', '2', '3', '4', '5', '6', '7',
52              '8', '9', 'A', 'B', 'C', 'D', 'E', 'F'};
53          char[] newChar = new char[b.length * 2];
54          for (int i = 0; i < b.length; i++) {
55              newChar[2 * i] = hex[(b[i] & 0xf0) >> 4];
56              newChar[2 * i + 1] = hex[b[i] & 0xf];
57          }
58          return new String(newChar);
59      }

61      public static byte[] hexString2ByteArray(String hexString) {
62          char[] chars = hexString.toCharArray();
63          byte[] b = new byte[chars.length / 2];
64          for (int i = 0; i < b.length; i++) {
65              int high = Character.digit(chars[2 * i], 16) << 4;
66              int low = Character.digit(chars[2 * i + 1], 16);
67              b[i] = (byte) (high | low);
68          }
69          return b;
70      }
