35      }

43      /**
44       * 利用递归的方式来组装树形结构
45       * @param infoType 
46       */
47      private void assembleInfoType(InfoType infoType) {
48          assembleInfoTypeContent(infoType);
49          for(InfoType child : infoType.getChild()){
50              //重点在这行代码，默认从XML解析出来的的树是向下引用，没有向上引用
51              //具体可看对应的XML配置文件即可明白
52              child.setParent(infoType);
53              assembleInfoType(child);
54          }
55      }
56      /**
57       * 默认从XML解析出来的的树有对内容的引用，但内容没有对树节点的引用
58       * @param infoType 
59       */
60      private void assembleInfoTypeContent(InfoType infoType) {
61          for(InfoTypeContent infoTypeContents : infoType.getInfoTypeContents()){
62              infoTypeContents.setInfoType(infoType);
63          }
64      }
