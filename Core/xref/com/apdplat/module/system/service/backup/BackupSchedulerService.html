22  @Service
23  public class BackupSchedulerService implements ApplicationListener {
24      protected static final APDPlatLogger log = new APDPlatLogger(BackupSchedulerService.class);
25  
26      private static SchedulerFactory sf = new StdSchedulerFactory();
27      @Resource(name = "serviceFacade")
28      protected ServiceFacade serviceFacade;
29      @Resource(name = "backupTask")
30      private JobDetail backupTask;
31  
32      @Override
33      public void onApplicationEvent(ApplicationEvent event) {
34          if (event instanceof ContextRefreshedEvent) {
35              log.info("spring容器初始化完成, 开始检查是否需要启动定时备份数据调度器");
36              BackupScheduleConfig config = getBackupScheduleConfig();
37              if (config != null && config.isEnabled()) {
38                  schedule(config.getScheduleHour(),config.getScheduleMinute());
39                  log.info("启动定时备份数据调度器");
40              }else{
41                  log.info("没有设置定时备份数据任务");
42              }
43          }
44      }
45      
46      public BackupScheduleConfig getBackupScheduleConfig(){        
47          Page<BackupScheduleConfig> page=serviceFacade.query(BackupScheduleConfig.class);
48          if(page.getTotalRecords()==1){
49              BackupScheduleConfig scheduleConfig=page.getModels().get(0);  
50              return scheduleConfig;
51          }
52          return null;
53      }
54  
55      public String unSchedule(){        
56          try {
57              BackupScheduleConfig config=getBackupScheduleConfig();
58              if(config!=null){
59                  config.setEnabled(false);
60                  serviceFacade.update(config);
61                  log.info("禁用定时重建配置对象");
62              }else{
63                  String tip="还没有设置定时备份数据任务";
64                  log.info(tip);
65                  return tip;
66              }
67              Scheduler sched = sf.getScheduler();
68              sched.deleteJob(backupTask.getName(), "DEFAULT");
69              String tip="删除定时备份数据任务，任务名为：" + backupTask.getName() + ",全名为: " + backupTask.getFullName();
70              log.info(tip);
71              return tip;
72          } catch (SchedulerException ex) {
73              String tip="删除定时备份数据任务失败，原因："+ex.getMessage();
74              log.info(tip);
75              return tip;
76          }
77      }

79      public String schedule(int hour, int minute) {
80          BackupScheduleConfig scheduleConfig = getBackupScheduleConfig();
81          if (scheduleConfig == null) {
82              //新建配置对象
83              BackupScheduleConfig config = new BackupScheduleConfig();
84              config.setScheduleHour(hour);
85              config.setScheduleMinute(minute);
86              config.setEnabled(true);
87              serviceFacade.create(config);
88          } else {
89              //修改配置对象
90              scheduleConfig.setScheduleHour(hour);
91              scheduleConfig.setScheduleMinute(minute);
92              scheduleConfig.setEnabled(true);
93              serviceFacade.update(scheduleConfig);
94          }
95  
96          String expression = "0 " + minute + " " + hour + " * * ?";
97          try {
98              CronExpression cronExpression = new CronExpression(expression);
99  
100             CronTrigger trigger = new CronTriggerBean();
101             trigger.setCronExpression(cronExpression);
102             trigger.setName("定时触发器,时间为：" + hour + ":" + minute);
103 
104             Scheduler sched = sf.getScheduler();
105             sched.deleteJob(backupTask.getName(), "DEFAULT");
106             sched.scheduleJob(backupTask, trigger);
107             sched.start();
108             String tip = "删除上一次的任务，任务名为：" + backupTask.getName() + ",全名为: " + backupTask.getFullName();
109             log.info(tip);
110             String taskState = "定时备份数据任务执行频率为每天，时间（24小时制）" + hour + ":" + minute;
111             log.info(taskState);
112             return taskState;
113         } catch (ParseException | SchedulerException e) {
114             String tip = "定时备份数据设置失败，原因：" + e.getMessage();
115             log.info(tip,e);
116             return tip;
117         }
118     }
