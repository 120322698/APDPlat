62              
63              String extendPropertyFiles = props.getProperty("extend.property.files");
64              if(extendPropertyFiles!=null && !"".equals(extendPropertyFiles.trim())){
65                  String[] files=extendPropertyFiles.trim().split(",");
66                  for(String file : files){

72                      }catch(Exception e){

75                      }      
76                  }
77              }    
78              log.info("系统配置属性装载完毕");
79              log.info("System configuration properties finished loading", Locale.ENGLISH);
80              log.info("******************属性列表***************************");

82              for(String propertyName : props.stringPropertyNames()){
