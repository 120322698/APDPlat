161                 LogQueue.getLogQueue().saveLog();
162         }
163         deregisterDrivers();
164         log.info("卸载JDBC驱动");
165         log.info("Uninstalled JDBC driver", Locale.ENGLISH);

171     private static void deregisterDrivers() {
172         Enumeration<Driver> drivers=DriverManager.getDrivers();
173         while(drivers.hasMoreElements()){
174             Driver driver=drivers.nextElement();
175             try {
176                 DriverManager.deregisterDriver(driver);
177             } catch (SQLException e) {
178                 log.warn("卸载JDBC驱动失败："+driver, e);
179                 log.warn("Fail to uninstall JDBC driver："+driver, e, Locale.ENGLISH);
180             }
181         }
182     }
183     private static void registerModules(){
184         StringBuilder modules=new StringBuilder();
185         try {
186             Enumeration<URL> ps = Thread.currentThread().getContextClassLoader().getResources("META-INF/services/module.xml");
187             while(ps.hasMoreElements()) {
188                 URL url=ps.nextElement();
189                 String file=url.getFile();
190                 if(file.contains(".jar!")){
191                     int start=file.indexOf("WEB-INF/lib/");
192                     int end=file.indexOf("!/META-INF/services/module.xml");
193                     if(start==-1 || end==-1){
194                         continue;
195                     }
196                     String jar=file.substring(start, end);
197                     modules.append(jar).append(",");
198                     log.info("注册模块："+jar);
199                     log.info("Register module："+jar, Locale.ENGLISH);
200                     extractWebFromModule(jar);
201                     extractDataFromModule(jar);
202                 }else{
203                     log.warn("在非jar包中找到META-INF/services/module.xml");
204                     log.warn("Find META-INF/services/module.xml in non-jar", Locale.ENGLISH);
205                 }
206             }
207         } catch (IOException e) {
208             log.error("注册模块出错", e);
209             log.error("Failed to register module", e, Locale.ENGLISH);
210         }
211         if(modules.length()>0){
212             modules=modules.deleteCharAt(modules.length()-1);
213         }
214         //从配置文件中获取属性
215         String scanJars=PropertyHolder.getProperty("scan.jars");
216         log.info("注册模块前，scanJars: "+scanJars);
217         log.info("Before register，scanJars: "+scanJars, Locale.ENGLISH);
218         if(scanJars!=null && !"".equals(scanJars.trim())){
219             scanJars=scanJars+","+modules.toString();
220         }else{
221             scanJars=modules.toString();
222         }
223         //设置到系统属性中
224         //spring会把系统属性中的配置覆盖掉配置文件中的配置
225         //<property name="systemPropertiesModeName" value="SYSTEM_PROPERTIES_MODE_OVERRIDE" />
226         System.setProperty("scan.jars", scanJars);
227         //设置回配置属性
228         PropertyHolder.setProperty("scan.jars", scanJars);
229         log.info("注册模块后，scanJars: "+scanJars);
230         log.info("After register，scanJars: "+scanJars, Locale.ENGLISH);
231     }
