107     @Override
108     public PropertyCriteria buildPropertyCriteria(){
109         PropertyCriteria propertyCriteria=super.buildPropertyCriteria();
110         if(propertyCriteria==null){
111             propertyCriteria=new PropertyCriteria();
112         }
113         //orgId==-1或orgId<0代表为根节点，不加过滤条件
114         if(orgId>0){
115             Org obj=service.retrieve(Org.class, orgId);
116             //获取orgId的所有子机构的ID
117             List<Integer> orgIds=OrgService.getChildIds(obj);
118             //加上orgId
119             orgIds.add(obj.getId());
120             
121             PropertyEditor pe=new PropertyEditor("org.id", Operator.in, PropertyType.List, orgIds);
122             propertyCriteria.addPropertyEditor(pe);
123             
124             return propertyCriteria;
125         }
126          
127         return propertyCriteria;

131         Integer[] ids=super.getIds();
132         if(ids!=null && ids.length>0){
133             if(!StringUtils.isBlank(password)){
134                 for(int id : ids){

150         int start=super.getStart();
151         int len=super.getLimit();
152         if(start==-1){
153             start=0;

155         if(len==-1){
156             len=10;
157         }
158         Org o=null;
159         Role r=null;
160         if(!StringUtils.isBlank(org)){
161             //返回特定组织架构及其所有子机构的在线用户
162             int id=Integer.parseInt(org);
163             o=service.retrieve(Org.class, id);
164         }
165         if(!StringUtils.isBlank(role)){
166             //返回属于特定角色的在线用户
167             int id=Integer.parseInt(role);
168             r=service.retrieve(Role.class, id);
169         }
170         
171         List<User> users=OnlineUserService.getUser(o,r);
172         log.info("获取在线用户,start: "+start+",len:"+len);
173         log.info("在线用户的总数为： "+users.size());
174         if(len>users.size()){
175             len=users.size();
176         }
177         List<User> models=new ArrayList<>();
178         for(int i=start;i<len;i++){
179             models.add(users.get(i));
180         }
181         //构造当前页面对象
182         page=new Page<>();
183         page.setModels(models);
184         page.setTotalRecords(users.size());
185         
186         Map json = new HashMap();
187         json.put("totalProperty", page.getTotalRecords());
188         List<Map> result = new ArrayList<>();
189         renderJsonForQuery(result);
190         json.put("root", result);
191         Struts2Utils.renderJson(json);
192         return null;

275         Map result=new HashMap();

277         if(user==null){
278             result.put("success", false);
279             result.put("message", "用户没有登录");
280              Struts2Utils.renderJson(result);
281             return null;

283         oldPassword=PasswordEncoder.encode(oldPassword.trim(),user);
284         if(oldPassword.equals(user.getPassword())){
285             user.setPassword(PasswordEncoder.encode(newPassword.trim(),user));
286             service.update(user);
287             result.put("success", true);
288             result.put("message", "修改成功");
289              Struts2Utils.renderJson(result);
