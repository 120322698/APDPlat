25      /**
26       * 对一段String生成MD5摘要信息
27       * @param message 要摘要的String
28       * @return 生成的MD5摘要信息
29       */
30      protected String getMD5(String message) {
31          message += "{apdplat}";
32          try {
33              MessageDigest md = MessageDigest.getInstance("MD5");
34              log.debug("MD5摘要长度：" + md.getDigestLength());
35              byte[] b = md.digest(message.getBytes("utf-8"));
36              String md5 = ConvertUtils.byte2HexString(b)+message.length();
37              return getSplitString(md5);
38          } catch (NoSuchAlgorithmException | UnsupportedEncodingException e) {
39              log.error("MD5摘要失败",e);
40          }
41          return null;
42      }

51      /**
52       * 将很长的字符串以固定的位数分割开，以便于人类阅读
53       * 如将
54       * 71F5DA7F495E7F706D47F3E63DC6349A
55       * 以-，每4个一组，则分割为
56       * 71F5-DA7F-495E-7F70-6D47-F3E6-3DC6-349A
57       * @param str 字符串
58       * @param split 分隔符
59       * @param length 长度
60       * @return 
61       */
62      protected String getSplitString(String str, String split, int length){        
63          int len=str.length();
64          StringBuilder temp=new StringBuilder();
65          for(int i=0;i<len;i++){
66              if(i%length==0 && i>0){
67                  temp.append(split);
68              }
69              temp.append(str.charAt(i));
70          }
71          String[] attrs=temp.toString().split(split);
72          StringBuilder finalMachineCode=new StringBuilder();
73          for(String attr : attrs){
74              if(attr.length()==length){
75                  finalMachineCode.append(attr).append(split);
76              }
77          }
78          String result=finalMachineCode.toString().substring(0, finalMachineCode.toString().length()-1);
79          return result;
80      }

97                  if (NetFlags.LOOPBACK_ADDRESS.equals(cfg.getAddress()) || (cfg.getFlags() & NetFlags.IFF_LOOPBACK) != 0
98                          || NetFlags.NULL_HWADDR.equals(cfg.getHwaddr())) {
99                      continue;
100                 }
101                 String mac = cfg.getHwaddr();
102                 result.add(mac);
103                 log.debug("mac: " + mac);
104             }
105             if(result.size()<1){
106                 return null;
107             }
108             Properties props = System.getProperties();
109             String javaVersion = props.getProperty("java.version");
110             result.add(javaVersion);
111             log.debug("Java的运行环境版本：    " + javaVersion);
112             String javaVMVersion = props.getProperty("java.vm.version");
113             result.add(javaVMVersion);
114             log.debug("Java的虚拟机实现版本：    " + props.getProperty("java.vm.version"));
115             String osVersion = props.getProperty("os.version");
116             result.add(osVersion);
117             log.debug("操作系统的版本：    " + props.getProperty("os.version"));
118 
119             Mem mem = sigar.getMem();
120             // 内存总量
121             String totalMem = mem.getTotal() / 1024L + "K av";
122             log.debug("内存总量:    " + totalMem);
123             result.add(totalMem);

126             String machineCode = getMD5(result.toString());
127
