64          //取交集

71              log.info("特定组织架构及其所有子组织架构:"+orgIds);
72              log.info("特定角色及其所有子角色:"+orgIds);
73              for(Object obj : users){

76                  if(orgIds.contains(user.getOrg().getId())){
77                      for(Role r : user.getRoles()){
