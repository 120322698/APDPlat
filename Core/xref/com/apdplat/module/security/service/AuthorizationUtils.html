24      private static String getSplitString(String str, String split, int length){        
25          int len=str.length();
26          StringBuilder temp=new StringBuilder();
27          for(int i=0;i<len;i++){
28              if(i%length==0 && i>0){
29                  temp.append(split);
30              }
31              temp.append(str.charAt(i));
32          }
33          String[] attrs=temp.toString().split(split);
34          StringBuilder finalMachineCode=new StringBuilder();
35          for(String attr : attrs){
36              if(attr.length()==length){
37                  finalMachineCode.append(attr).append(split);
38              }
39          }
40          String result=finalMachineCode.toString().substring(0, finalMachineCode.toString().length()-1);
41          return result;
42      }
