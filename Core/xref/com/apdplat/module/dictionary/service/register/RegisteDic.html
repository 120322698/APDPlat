41      private void registeDic(Dic dic){
42          Page<Dic> page=serviceFacade.query(Dic.class);
43          if(page.getTotalRecords()==0){
44              log.info("第一次注册第一个数据字典(First time to register the first data dictionary): "+dic.getChinese());            
45              serviceFacade.create(dic);
46              //保存根数据字典
47              rootDic=dic;
48              data.add(dic);
49          }else{
50              log.info("以前已经注册过数据字典(Already registered the data dictionary)");
51              log.info("查找出根数据字典(Find out the root data dictionary)");
52              Dic root=null;
53              Dic existRoot=dicService.getRootDic();
54              if(existRoot!=null){
55                  root=existRoot;
56                  log.info("找到以前导入的根数据字典(Find out previous imported root data dictionary): "+root.getChinese());
57              }else{
58                  log.info("没有找到以前导入的根数据字典(Failed to find out previous imported root data dictionary)");
59                  if(rootDic!=null){
60                      log.info("使用本次导入的根数据字典(Use the import root data dictionary)");
61                      root=rootDic;
62                  }
63              }
64              if(root!=null){
65                  log.info("将第一次以后的数据字典的根数据字典设置为第一次注册的根数据字典(Set the later data dictionary to the first registered root data dictionary)");
66                  for(Dic subDic : dic.getSubDics()){
67                      if(hasRegisteDic(subDic)){
68                          log.info("数据字典(This data dictionary) "+subDic.getChinese()+" (already registered)在此前已经被注册过，此次忽略，检查其子数据字典");
69                          registeSubDic(subDic);
70                          continue;
71                      }
72                      subDic.setParentDic(root);
73                      log.info("注册后续数据字典(Register the follow-up data dictionary): "+subDic.getChinese());
74                      serviceFacade.create(subDic);
75                      data.add(subDic);
76                  }
77              }else{
78                  log.info("没有找到根数据字典，注册失败！(Not find out root data dictionary, failed to register)");
79              }
80          }
81      }
82      private void registeSubDic(Dic dic){
83          for(Dic sub : dic.getSubDics()){
84              if(hasRegisteDic(sub)){
85                  log.info("数据字典(Data dictionary) "+sub.getChinese()+" (already registered, checking other sub-data dictionary)在此前已经被注册过，此次忽略，检查其子数据字典");
86                  registeSubDic(sub);
87              }else{
88                  log.info("注册后续数据字典(Register follow-up data dictionary): "+sub.getChinese());
89                  //重新从数据库中查询出数据字典，参数中的数据字典是从XML中解析出来的
90                  sub.setParentDic(dicService.getDic(dic.getEnglish()));
91                  serviceFacade.create(sub);
92                  data.add(sub);
93              }
94          }
95      }
96      private boolean hasRegisteDic(Dic dic){
97          Dic existsDic=dicService.getDic(dic.getEnglish());
98          if(existsDic==null){
99              return false;
100         }
101         return true;
102     }
