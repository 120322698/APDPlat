31      public static List<Module> getRootModules(){
32          log.info("module.disable:"+disableModules);
33          List<Module> modules=new ArrayList<>();
34          try{
35              Enumeration<URL> ps = Thread.currentThread().getContextClassLoader().getResources("META-INF/services/module.xml");
36              while(ps.hasMoreElements()) {
37                  InputStream in = null;
38                  try {
39                      URL url=ps.nextElement();

43                      modules.add(root);
44                  }catch(Exception e)
45                  {
46                      log.error("获取根模块出错",e);
47                  }
48                  finally {
49                      try {
50                          in.close();

62      /**
63       * 解析并获得根模块
64       * @param in
65       * @return 
66       */
67      public static Module getRootModule(InputStream in){
68          //准备模块DTD文件以供校验使用
69          prepareDtd();
70          try {
71              byte[] data=FileUtils.readAll(in);
72              String xml=new String(data,"utf-8");
73              log.info("将DTD文件替换为绝对路径");
74              xml=xml.replace("module.dtd", FileUtils.getAbsolutePath(dtdFile));
75              log.info("将模块文件读取到字节数组中，长度为："+data.length);
76              ByteArrayInputStream bin=new ByteArrayInputStream(xml.getBytes("utf-8"));    
77              log.info("注册module.xml文件");
78              log.info("验证module.xml文件");
79              //校验文件
80              verifyFile(bin);
81              // 解析模块
82              Module module=parseModule(xml);
83              return module;
84          } catch (UnsupportedEncodingException e) {
85              log.error("获取根模块出错",e);
86          }
87          return null;
88      }
89      private static Module parseModule(String xml) {
90          XMLFactory factory=new XMLFactory(Module.class,Command.class);
91          Module module=factory.unmarshal(xml);
92          //将XML表示的模块转变为JAVA对象表示的模块
93          assembleMudule(module);
94          return module;
95      }
96      /**
97       * 采用递归的方式建立模块树的引用关系
98       * @param module  模块树的根
99       */
100     private static void assembleMudule(Module module){
101         int order=1;
102         for(Command c : module.getCommands()){
103             c.setModule(module);
104             if(c.getOrderNum()==0){
105                 c.setOrderNum(order++);
106             }
107         }
108         order=1;
109         List<Module> toDelete=new ArrayList<>();
110         for(Module m : module.getSubModules()){            
111             //根据参数module.hide来设置模块
112             if(disableModules.contains(m.getEnglish())){
113                 log.info("禁用: "+m.getChinese());
114                 toDelete.add(m);                
115                 continue;
116             }else{
117                 log.info("启用: "+m.getChinese());
118             }
119             m.setParentModule(module);
120             if(m.getOrderNum()==0){
121                 m.setOrderNum(order++);
122             }
123             assembleMudule(m);
124         }
125         for(Module m : toDelete){
126             module.removeSubModule(m);
127         }
128     }
