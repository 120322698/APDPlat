53      /**
54       * 获取某个路径下的指定的Package下的所有类，不包括<code>outsides</code>中的
55       */
56      public static List<Class<?>> getClasses(File dir, String pk, String[] outsides) throws ClassNotFoundException {
57          log.debug("  Dir: {}, PK: {}", new Object[]{dir, pk});
58          List<Class<?>> classes = new ArrayList<>();
59          if (!dir.exists()) {
60              return classes;
61          }
62          String thisPk = StringUtils.isBlank(pk) ? "" : pk + ".";
63          for (File f : dir.listFiles()) {
64              if (f.isDirectory()) {
65                  classes.addAll(getClasses(f, thisPk + f.getName(), outsides));
66              }
67              String name = f.getName();
68              if (name.endsWith(".class")) {
69                  Class<?> clazz = null;
70                  String clazzName = thisPk + name.substring(0, name.length() - 6);
71                  log.debug("Class: {}", clazzName);
72                  if (outsides == null || outsides.length == 0 || !ArrayUtils.contains(outsides, clazzName)) {
73                      try {
74                          clazz = Class.forName(clazzName);
75                      } catch (Throwable e) {
76                          log.error("实例化失败",e);
77                      }
78                      if (clazz != null) {
79                          classes.add(clazz);
80                      }
81                  }
82              }
83          }
84          return classes;
85      }

174     /**
175      * 通过反射,获得定义Class时声明的父类的泛型参数的类型.
176      *
177      * @param clazz clazz The class to introspect
178      * @param index the Index of the generic ddeclaration,start from 0.
179      * @return the index generic declaration, or Object.class if cannot be determined
180      */
181     @SuppressWarnings("unchecked")
182     public static Class getSuperClassGenricType(final Class clazz, final int index) {
183 
184         Type genType = clazz.getGenericSuperclass();
185 
186         if (!(genType instanceof ParameterizedType)) {
187             log.warn(clazz.getSimpleName() + "'s superclass not ParameterizedType");
188             return Object.class;
189         }
190 
191         Type[] params = ((ParameterizedType) genType).getActualTypeArguments();
192 
193         if (index >= params.length || index < 0) {
194             log.warn("Index: " + index + ", Size of " + clazz.getSimpleName() + "'s Parameterized Type: "
195                     + params.length);
196             return Object.class;
197         }
198         if (!(params[index] instanceof Class)) {
199             log.warn(clazz.getSimpleName() + " not set the actual class on superclass generic parameter");
200             return Object.class;
201         }
202         return (Class) params[index];
203     }

243     public static Field getDeclaredField(final Object object, final String fieldName) {
244         Assert.notNull(object, "object不能为空");
245         Assert.hasText(fieldName, "fieldName");
246         for (Class<?> superClass = object.getClass(); superClass != Object.class; superClass = superClass.getSuperclass()) {
247             try {
248                 return superClass.getDeclaredField(fieldName);
249             } catch (NoSuchFieldException e) {
250                 // Field不在当前类定义,继续向上转型
251                 //e.printStackTrace();
252             }
253         }
254         return null;

256     public static List<Field> getDeclaredFields(final Object object) {
257         Assert.notNull(object, "object不能为空");
258         List<Field> fields=new ArrayList<>();
259         for (Class<?> superClass = object.getClass(); superClass != Object.class; superClass = superClass.getSuperclass()) {
260             Field[] f=superClass.getDeclaredFields();
