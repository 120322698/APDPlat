180   /**
181    * 得到格式化的大小。
182    * @param size 原始大小
183    * @param unit 单位
184    * @return 根据单位进行格式化后的字符串大小
185    * 
186    */
187   public String format(long size, int unit) {
188     String result;
189     switch (unit) {
190       case BYTE:
191         result = formatter.format(size);
192         break;
193       case KILO:
194         result = formatter.format( ( (double) size) / ( (double) KILOBYTE));
195         break;
196       case MEGA:
197         result = formatter.format( ( (double) size) / ( (double) MEGABYTE));
198         break;
199       case GIGA:
200         result = formatter.format( ( (double) size) / ( (double) GIGABYTE));
201         break;
202       default:
203         result = formatter.format(size);
204         break;
205     }
206     if (showUnitName == true) {
207       result += unitNames[unit];
208     }
209     return result;
210   }
