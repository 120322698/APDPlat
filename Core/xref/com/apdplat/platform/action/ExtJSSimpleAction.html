163     public String updatePart() {
164         try{
165             Integer version=model.getVersion();
166             //此时的model里面存的值是从浏览器传输过来的
167             List<Property> properties=getPartProperties(model);
168             //此时的model里面存的值是从数据库里面加载的
169             model=service.retrieve(modelClass,model.getId());
170             
171             //数据版本控制，防止多个用户同时修改一条数据，造成更新丢失问题
172             if(version==null){
173                 log.info("前台界面没有传递版本信息");
174                 throw new RuntimeException("您的数据没有版本信息");
175             }else{
176                 log.info("前台界面传递了版本信息,version="+version);
177             }
178             if(version!=model.getVersion()){
179                 log.info("当前数据的版本为 "+model.getVersion()+",您的版本为 "+version);
180                 throw new RuntimeException("您的数据已过期，请重新修改");
181             }
182             
183             old(model);
184             for(Property property : properties){
185                 //把从浏览器传来的值射入model
186                 if(property.getName().contains(".")){
187                      //处理两个对象之间的引用，如：model.org.id=1
188                      if(property.getName().contains(".id")){
189                         String[] attr=property.getName().replace(".",",").split(",");
190                         if(attr.length==2){
191                             Field field=ReflectionUtils.getDeclaredField(model, attr[0]);
192                             T change=service.retrieve((Class<T>)field.getType(), (Integer)property.getValue());
193                             ReflectionUtils.setFieldValue(model, attr[0], change);
194                         }
195                      }
196                 }
197                 else{
198                     ReflectionUtils.setFieldValue(model, property.getName(), property.getValue());
199                 }
200             }
201             now(model);
202             //在更新前调用模板方法对模型进行处理
203             assemblyModelForUpdate(model);
204             service.update(model);
205             afterSuccessPartUpdateModel(model);
206         }catch(Exception e){
207             log.error("更新模型失败",e);
208             afterFailPartUpdateModel(model);
209             map=new HashMap();
210             map.put("success", false);
211             map.put("message", "修改失败 "+e.getMessage());
212             Struts2Utils.renderJson(map);
213             return null;
214         }
215         map=new HashMap();
216         map.put("success", true);
217         map.put("message", "修改成功");
218         Struts2Utils.renderJson(map);
219         return null;
220     }

494     private void addFieldValue(T obj, Field field, Map<String,String> data){
495         String fieldName = field.getName();
496         try{
497             if(field.isAnnotationPresent(Lob.class)){
498                 log.debug("字段["+fieldName+"]为大对象，忽略生成JSON字段");
499                 return;
500             }
501             Object value = ReflectionUtils.getFieldValue(obj, field);
502             if(value==null){
503                 data.put(fieldName, "");
504                 return;
505             }
506             //处理集合类型
507             if(field.isAnnotationPresent(ModelCollRef.class)){
508                 ModelCollRef ref = field.getAnnotation(ModelCollRef.class);
509                 String fieldRef = ref.value();
510                 Collection col=(Collection)value;
511                 String colStr="";
512                 if(col!=null){
513                     log.debug("处理集合,字段为："+field.getName()+",大小为："+col.size());
514                     if(col.size()>0){
515                         StringBuilder str=new StringBuilder();
516                         for(Object m : col){
517                             str.append(ReflectionUtils.getFieldValue(m, fieldRef).toString()).append(",");
518                         }
519                         str=str.deleteCharAt(str.length()-1);
520                         colStr=str.toString();
521                     }
522                 }else{
523                     log.debug("处理集合失败，"+value+" 不能转换为集合");
524                 }
525                 data.put(fieldName, colStr);
526                 return ;
527             }
528             //处理复杂对象类型
529             if(field.isAnnotationPresent(ModelAttrRef.class)){
530                 log.debug("处理对象,字段为："+field.getName());
531                 ModelAttrRef ref = field.getAnnotation(ModelAttrRef.class);
532                 String fieldRef = ref.value();
533                 //加入复杂对象的ID
534                 Object id = ReflectionUtils.getFieldValue(value, "id");
535                 data.put(fieldName+"_id", id.toString());
536                 //因为是复杂对象，所以变换字段名称
537                 fieldName=fieldName+"_"+fieldRef;
538                 //获取fieldRef的值
539                 value = ReflectionUtils.getFieldValue(value, fieldRef);
540             }
541             if(value.getClass()==null){
542                 data.put(fieldName, "");
543                 return;
544             }
545             String valueClass=value.getClass().getSimpleName();
546 
547             if("PersistentBag".equals(valueClass)){
548                 value="";
549             }
550             if("Timestamp".equals(valueClass) || "Date".equals(valueClass)){
551                 value=DateTypeConverter.toDefaultDateTime((Date)value);
552             }
553             //处理下拉菜单
554             if("DicItem".equals(valueClass)){
555                 value = ReflectionUtils.getFieldValue(value, "name");
556             }
557             data.put(fieldName, value.toString());
558         }catch(Exception e){
559             log.error("获取字段值失败",e);
560         }

586     protected void objectReference(T model) {
587         Field[] fields = model.getClass().getDeclaredFields();//获得对象方法集合
588         for (Field field : fields) {// 遍历该数组
589             if(field.isAnnotationPresent(ManyToOne.class) || field.isAnnotationPresent(OneToOne.class)){
590                 log.debug(model.getMetaData()+" 有ManyToOne 或 OneToOne映射，字段为"+field.getName());
591                 Model value=(Model)ReflectionUtils.getFieldValue(model, field);
592                 if(value==null){
593                     log.debug(model.getMetaData()+" 的字段"+field.getName()+"没有值，忽略处理");
594                     continue;
595                 }
596                 int id=value.getId();
597                 log.debug("id: "+id);
598                 value=service.retrieve(value.getClass(), id);
599                 ReflectionUtils.setFieldValue(model, field, value);
600             }
601         }
