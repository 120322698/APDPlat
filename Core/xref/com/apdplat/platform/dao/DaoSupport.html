105     private void bindingPropertyCriteria(Query query, PropertyCriteria propertyCriteria) {
106         if (query != null && propertyCriteria != null) {
107             List<PropertyEditor> propertyEditors=propertyCriteria.getPropertyEditors();
108             int len=propertyEditors.size();
109             for (int i=0;i<len;i++) {
110                 PropertyEditor propertyEditor = propertyEditors.get(i);
111                 List<PropertyEditor> subPropertyEditor=propertyEditor.getSubPropertyEditor();
112                 if(subPropertyEditor==null || subPropertyEditor.isEmpty()){
113                     query.setParameter(propertyEditor.getProperty().getNameParameter(), propertyEditor.getProperty().getValue());
114                 }else{
115                     binding(query,propertyEditor,1);
116                 }
117             }
118         }
119     }

121     private void binding(Query query, PropertyEditor propertyEditor,int level) {
122             List<PropertyEditor> subPropertyEditor=propertyEditor.getSubPropertyEditor();
123 
124             int l=subPropertyEditor.size();
125             for(int j=0;j<l;j++){
126                 PropertyEditor p = subPropertyEditor.get(j);
127                 List<PropertyEditor> ss=p.getSubPropertyEditor();
128                 if(ss==null || ss.isEmpty()){
129                     query.setParameter(p.getProperty().getNameParameter()+"_"+level+"_"+j, p.getProperty().getValue());
130                 }else{
131                     binding(query,p,++level);
132                 }
133             }
134     }

149     /**
150      * 组装where 语句
151      * @param propertyCriteria
152      * @return
153      */
154     private <T extends Model> String buildPropertyCriteria(PropertyCriteria propertyCriteria) {
155         StringBuilder wherejpql = new StringBuilder("");
156         String result = "";
157         if (propertyCriteria != null && propertyCriteria.getPropertyEditors().size() > 0) {
158             //判断是否支持集合查询
159             if(propertyCriteria.getCollection()!=null && propertyCriteria.getObject()!=null){
160                 wherejpql.append(" join o.").append(propertyCriteria.getCollection()).append(" ").append(propertyCriteria.getObject());
161             }
162             
163             wherejpql.append(" where ");
164 
165             List<PropertyEditor> propertyEditors=propertyCriteria.getPropertyEditors();
166             int len=propertyEditors.size();
167             for (int i=0;i<len;i++) {
168                 PropertyEditor propertyEditor=propertyEditors.get(i);
169                 List<PropertyEditor> subPropertyEditor=propertyEditor.getSubPropertyEditor();
170                 //当没有子属性的时候时，属性编辑器本身才是有效的，否则他就是子属性的一个容器而已
171                 if(subPropertyEditor==null || subPropertyEditor.isEmpty()){
172                     //判断是否支持集合查询
173                     if(propertyCriteria.getCollection()!=null && propertyCriteria.getObject()!=null && propertyEditor.getProperty().getName().startsWith(propertyCriteria.getObject())){
174                         wherejpql.append(" ");
175                     }else{
176                         wherejpql.append(" o.");
177                     }
178                     wherejpql.append(propertyEditor.getProperty().getName())
179                             .append(" ")
180                             .append(propertyEditor.getPropertyOperator().getSymbol())
181                             .append(" ")
182                             .append(":")
183                             .append(propertyEditor.getProperty().getNameParameter());
184                     if(i<len-1){
185                             wherejpql.append(" ");
186                             wherejpql.append(propertyCriteria.getCriteria().name());
187                             wherejpql.append(" ");
188                     }
189                 }
190                 else{
191                     wherejpql.append(dealWithSubPropertyEditor(propertyEditor,1));
192                 }
193             }
194             result = wherejpql.toString();
195         }
196         return result;
197     }
198     private String dealWithSubPropertyEditor(PropertyEditor propertyEditor,int level){
199             StringBuilder wherejpql=new StringBuilder();
200             List<PropertyEditor> subPropertyEditor=propertyEditor.getSubPropertyEditor();
201             wherejpql.append(" ( ");
202             for (int j=0;j<subPropertyEditor.size();j++) {
203                 PropertyEditor sub = subPropertyEditor.get(j);
204                 List<PropertyEditor> ss=sub.getSubPropertyEditor();
205                 //当没有子属性的时候时，属性编辑器本身才是有效的，否则他就是子属性的一个容器而已
206                 if(ss!=null && !ss.isEmpty()){
207                     wherejpql.append(dealWithSubPropertyEditor(sub,++level));
208                 }
209                 else{
210                     wherejpql.append(" o.").append(sub.getProperty().getName()).append(" ").append(sub.getPropertyOperator().getSymbol()).append(" ")
211                             .append(":")
212                             .append(sub.getProperty().getNameParameter())
213                             .append("_")
214                             .append(level)
215                             .append("_")
216                             .append(j);
217                 }
218                 if(j<subPropertyEditor.size()-1){
219                     wherejpql.append(" ").append(propertyEditor.getCriteria().name()).append(" ");
220                 }
221             }
222             wherejpql.append(" ) ");
223             return wherejpql.toString();
224     }
225     /**
226      * 组装order by语句
227      * @param sortCriteria
228      * @return
229      */
230     private String buildOrderCriteria(OrderCriteria sortCriteria) {
231         StringBuilder orderbyql = new StringBuilder("");
232         if (sortCriteria != null && sortCriteria.getOrders().size() > 0) {
233             orderbyql.append(" order by ");
234 
235             for (Order order : sortCriteria.getOrders()) {
236                 orderbyql.append("o.").append(order.getPropertyName()).append(" ").append(order.getSequence().getValue()).append(",");
237             }
238             orderbyql.deleteCharAt(orderbyql.length() - 1);
239         }
240         return orderbyql.toString();
241     }

313     private <T extends Model> T hightlight(Class<T> modelClass, CompassHits hits, int i) {
314         T model = (T) hits.data(i);
315         //在对模型进行高亮之前先从数据库加载模型
316         model = em.find(modelClass, model.getId());
317         //防止高亮设置更新进入文档数据
318         em.detach(model);
319         CompassHighlighter highlighter = hits.highlighter(i);
320         //高亮处理
321         for (String searchProperty : model.getSearchProperties()) {
322             try {
323                 String value = highlighter.fragment(searchProperty);
324                 if (StringUtils.isNotBlank(value)) {
325                     try {
326                         ReflectionUtils.setFieldValue(model, searchProperty, value);
327                         log.debug("给对象【" + model.getMetaData() +" : "+model.getId()+ "】的【" + searchProperty + "】属性添加高亮成功");
328                     } catch (Exception e) {
329                         log.debug("添加高亮，给对象【" + model.getMetaData() + "】设置属性【" + searchProperty + "】失败,值为:【" + value + "】");
330                     }
331                 }
332             } catch (Exception e) {
333                 log.debug("处理" + searchProperty + "高亮抛出异常，忽略");
334             }
335         }
336         return model;
337     }
