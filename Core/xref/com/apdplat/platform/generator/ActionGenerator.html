85              int index=actionPackageBase.indexOf("\\src\\main\\java\\");
86              actionPackageBase=actionPackageBase.substring(index).replace("\\src\\main\\java\\", "").replace("\\", ".");
87              for(Module module : modules){        
88                  List<Command> specialCommands=ModuleService.getSpecialCommand(module);    
89                  String action=Character.toUpperCase(module.getEnglish().charAt(0))+module.getEnglish().substring(1);
90                  String actionName=action+"Action";    
91                  String actionNamespace=ModuleService.getModulePath(module.getParentModule());
92                  String actionPackage=actionPackageBase+"."+actionNamespace.replace("/", ".").replace("\\", ".")+"action";
93                  actionNamespace=actionNamespace.substring(0, actionNamespace.length()-1);
94                  String actionPath=actionPackage.replace(".", "/");
95                  //判断此模块是否已被指定操作特定的模型
96                  Model model=actionToModel.get(module.getEnglish());
97                  System.out.println("actionToModel action: "+module.getEnglish());
98                  if(model!=null){
99                      System.out.println("actionToModel model: "+model.getClass().getName());
100                     System.out.println("模块 "+module.getChinese()+" 已被指定操作特定的模型 "+model.getMetaData());
101                     //模块有指定的Model
102                     String modelPackage=model.getClass().getName().replace("."+model.getClass().getSimpleName(), "");
103                     generateAction(actionPackage,actionNamespace,actionName,modelPackage,model.getClass().getSimpleName(),workspaceModuleBasePath,specialCommands);

106                 System.out.println("模块 "+module.getChinese()+" 没有对应的模型 ");
107                 
108                 if(models.contains(action)){

157     /**
158      * 找到model文件夹
159      * @param file
160      * @return 
161      */
162     private static File findModel(File file){
163         //查看当前文件夹是否为 model 文件夹
164         if(file.getName().equals("model")){
165             return file;
166         }else{
167             File[] subFiles=file.listFiles(new FilenameFilter(){
168 
169                 @Override
170                 public boolean accept(File dir, String name) {
171                     if(name.startsWith(".")){
172                         return false;
173                     }
174                     return true;
175                 }
176                 
177             });
178             if(subFiles==null){
179                 return null;
180             }
181             //对当前文件夹的子文件夹进行判断
182             for(File child : subFiles){
183                if(child.getName().equals("model")){
184                     return child;
185                 } 
186             }
187             //当前文件夹的子文件夹没有model文件夹
188             for(File child : subFiles){
189                 File result=findModel(child);
190                 if(result!=null){
191                     return result;
192                 }
193             }            
194         }
195         return null;
196     }

203         String p=clazz.getPackage().getName();
204         p=p.replace(".model","");
205         int index=p.lastIndexOf(".");
206         String _package=p.substring(0,index);
207         String actionNamespace=p.substring(index+1);
208         String model=clazz.getSimpleName();
209         String actionName=model+"Action";
210         String actionPackage=_package+"."+actionNamespace+".action";
211         String modelPackage=_package+"."+actionNamespace+".model";
212         
213         //普通情况下一个模型对应一个Action，如果某些模型不需要Action（即没有在module.xml中进行声明），则忽略生成此模型对应的Action
214         //从module.xml中查找该model是否配置在模块文件中
215         String shortModel=Character.toLowerCase(model.charAt(0))+model.substring(1);
216         Module m=ModuleService.getModuleFromXml(shortModel);
217         if(m==null){
218             log.info(shortModel+" 没有在module.xml中进行声明，忽略生成Action");
219             return ;
220         }        
221         //添加自定义的方法
222         List<Command> specialCommands=ModuleService.getSpecialCommand(m);
223         System.out.println("generateFromModel actionPackage："+actionPackage);
224         System.out.println("generateFromModel actionNamespace："+actionNamespace);
225         System.out.println("generateFromModel model："+model);
226         
227         generateAction(actionPackage,actionNamespace,actionName,modelPackage,model,workspaceModuleBasePath,specialCommands);
228     }

284             }
285             File file = new File(workspaceModuleBasePath);
286             file = new File(file, actionPath);
287             if (!file.exists()) {
288                 file.mkdirs();
289             }
290             file = new File(file, actionName+".java");
291             if (!file.exists()) {
292                 try {
293                     file.createNewFile();
294                 } catch (IOException e) {
295                     log.error("生成ACTION错误",e);
296                 }
297             }else{
298                 log.info("源文件已经存在，请删除 "+file.getAbsolutePath()+" 后在执行命令");

300             }
301             saveFile(file,content);
